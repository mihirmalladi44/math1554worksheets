    \begin{equation*} Q(\vec{x}) = \vec{x}^T A \vec{x} = \begin{pmatrix} x_1 \amp x_2 \amp \ldots \amp x_n \end{pmatrix} \begin{pmatrix} a_{11} \amp a_{12} \amp \ldots \amp a_{1n} \\ a_{12} \amp a_{22} \amp \ldots \amp a_{2n} \\ \vdots \amp \vdots \amp \ddots \amp \vdots \\ a_{1n} \amp a_{2n} \amp \ldots \amp a_{nn} \end{pmatrix} \begin{pmatrix} x_1 \\ x_2 \\ \vdots \\ x_n \end{pmatrix} \end{equation*}
